#### Extension 1: Failure Analyzer

- Compares with past failures

```java
@Extension
public class FailureAnalyzerExtension implements McpServerExtension {
    
    @Tool(description = "Analyzes a failed build and returns structured failure data")
    public FailureAnalysis analyzeBuildFailure(
        @ToolParam(description = "Job name") String jobName,
        @ToolParam(description = "Build number") Integer buildNumber
    ) {
        // 1. Get build
        Run build = getBuild(jobName, buildNumber);
        
        // 2. Get logs
        String logs = getConsoleLogs(build);
        
        // 3. Extract errors
        List<ErrorInfo> errors = extractErrors(logs);
        
        // 4. Get recent commits
        List<CommitInfo> commits = getRecentCommits(build);
        
        // 5. Find similar past failures
        List<SimilarFailure> similar = findSimilarFailures(jobName, errors);
        
        // 6. Return structured data
        return new FailureAnalysis(
            build.getResult().toString(),
            errors,
            commits,
            similar,
            logs.substring(Math.max(0, logs.length() - 5000)) // Last 5000 chars
        );
    }
}
```
